import { DontBlockItem } from '../components/DoVsDont/DontBlock';

## Common Do's and Don'ts

<DoVsDontRow>
  {/* prettier-ignore */}
  <DoBlockItem
    title='Use className for layout positioning'
    description='Only use className for layout-related utilities (positioning, margins, display properties).'
  >
    ```tsx
    <Component className='mt-4 absolute top-0 md:hidden'>
        Submit
    </Component>
    ```
  </DoBlockItem>
  {/* prettier-ignore */}
  <DontBlockItem
    title='ClassName is not allowed for core styling'
    description='Core styling, internal spacing, and typography cannot be modified using className'
  >
    ```tsx
    <Component className='bg-accent'>
      Submit
    </Component>
    ```
  </DontBlockItem>
</DoVsDontRow>

<DoVsDontRow>

{/* prettier-ignore */}
<DoBlockItem
  title='Import specific icons'
  description='Use specific icons for better performance and tree-shaking'
>
  ```tsx 
  import { Check } from '@ledgerhq/lumen-ui-rnative/symbols';

  <Component icon={Check}> Label</Component>
  ```
</DoBlockItem>

  <DontBlockItem
    title="Don't import all icons"
    description='Avoid importing entire icon set for better bundle size'
  >
    ```tsx 
    import * as Icons from '@ledgerhq/lumen-ui-rnative/symbols';
    
    <Component icon={Icons.Check}> Label</Component>
    ```
  </DontBlockItem>
</DoVsDontRow>
